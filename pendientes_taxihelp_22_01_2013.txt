<?php
class Zend_View_Helper_MsgInAs extends Zend_View_Helper_Abstract 
{  
    public function msgInAs ($message)
    {
        $auth = Zend_Auth::getInstance();  
        if ($auth->hasIdentity()) {            
            $mdlGen = new Application_Model_Generales();
            $mdlRes = new Application_Model_Reserva();
            $respen=substr($message, 0 , strlen($message)-1);
            $arr_vales=$mdlGen->_StringaArray($respen);
            $posts = $mdlRes->ListarResTime2($arr_vales, '1');
                if(count($posts)>0){ $i=1;
                    $html='<div class="notify">';
                    foreach ($posts as $x):   
                     if($i<=3){
                    $html.='<div class="wrap_noti alerta notify">';
                                    	$html.='<div class="fleft cont">';
                                        	$html.='<div class="fleft text">';
                                            	$html.='<span>'.$x->cliente.' <br /> Nro Doc. '.$x->nro_documento.'</span>';
                                                $html.='<br>Observacion:'.$x->obs.'<br />';
                                                $html.='<br class="fec">Fecha recojo: <br />'.$x->fecha_recojo.' - '.$x->hora_recojo.'<br />';
                                                $html.='</div>';
                                        $html.='</div>';
                    $html.='</div>'; $i++;}
                    endforeach;
                    $html.='</div>';  
                } 
            }
            return $html; 
    } 
} ?>
